import smtplib from email.mime.text import MIMEText from email.mime.multipart import MIMEMultipart def send_email(to_email, token): # create the email message msg = MIMEMultipart() msg['From'] = 'your_email@example.com' msg['To'] = to_email msg['Subject'] = 'Reset your password' # create the email body body = f'Hi,\n\nPlease use the following token to reset your password: {token}\n\nBest regards,\nYour website team' msg.attach(MIMEText(body, 'plain')) # send the email with smtplib.SMTP('smtp.gmail.com', 587) as smtp: smtp.starttls() smtp.login('your_email@example.com', 'your_email_password') smtp.send_message(msg) # Example usage: send_email('user@example.com', 'abcdefg')
